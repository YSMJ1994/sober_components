折叠容器的基本用法

```typescript jsx
import React, { useState } from "react";
import { CollapseBox } from "sober_components";
import "sober_components/CollapseBox/style/index.scss";

export default function demo() {
  const [show, setShow] = useState<boolean>(false);
  const [show2, setShow2] = useState<boolean>(false);
  return (
    <div>
      <button onClick={() => setShow(!show)}>{show ? "隐藏" : "显示"}</button>
      <button onClick={() => setShow2(!show2)}>
        {show2 ? "隐藏" : "显示"}
      </button>
      <CollapseBox show={show}>
        <CollapseBox show={show2}>
          <h1>fafafsaf</h1>
        </CollapseBox>
        <h1>标题</h1>
        <p>
          测试内容测试内容测试内容测试内容测试内容测试内容测试内容测试内容测试内容测试内容
        </p>
      </CollapseBox>
    </div>
  );
}
```
